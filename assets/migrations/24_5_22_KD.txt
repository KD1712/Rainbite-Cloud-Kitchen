CREATE TABLE `rainbite`.`listing` ( `Id` INT(10) NOT NULL AUTO_INCREMENT , `R_Name` VARCHAR(255) NOT NULL , `Category` VARCHAR(255) NOT NULL , `S_Type` ENUM('Delivery','Take Away','Both') NOT NULL , `Phone_No` VARCHAR(10) NOT NULL , `Email` VARCHAR(40) NOT NULL , `City` VARCHAR(40) NOT NULL , `Address` VARCHAR(312) NOT NULL , `Pincode` VARCHAR(10) NOT NULL , PRIMARY KEY (`Id`)) ENGINE = InnoDB;
